035public class PinPadTerminal implements AutoCloseable {
036        private static final int CM_IOCTL_GET_FEATURE_REQUEST = CARD_CTL_CODE(3400);
037
038        private Map<FEATURE, Integer> features = new HashMap<PinPadTerminal.FEATURE, Integer>();
039        private boolean display = false;
040        private CardTerminal t = null;
041        private Card c = null;

043        // IOCTL-s of this terminal
044        public static enum FEATURE {
045                VERIFY_PIN_START(0x01),
046                VERIFY_PIN_FINISH(0x02),
047                MODIFY_PIN_START(0x03),
048                MODIFY_PIN_FINISH(0x04),
049                GET_KEY_PRESSED(0x05),
050                VERIFY_PIN_DIRECT(0x06),
051                MODIFY_PIN_DIRECT(0x07),
052                MCT_READERDIRECT(0x08),
053                MCT_UNIVERSAL(0x09),
054                IFD_PIN_PROPERTIES(0x0A),
055                ABORT(0x0B),
056                SET_SPE_MESSAGE(0x0C),
057                VERIFY_PIN_DIRECT_APP_ID(0x0D),
058                MODIFY_PIN_DIRECT_APP_ID(0x0E),
059                WRITE_DISPLAY(0x0F),
060                GET_KEY(0x10),
061                IFD_DISPLAY_PROPERTIES(0x11),
062                GET_TLV_PROPERTIES(0x12),
063                CCID_ESC_COMMAND(0x13),
064                EXECUTE_PACE(0x20);
065
066                private final int value;
067
068                private FEATURE(int value) {
069                        this.value = value;
070                }
071
072                public int getValue() {
073                        return value;
074                }
075                public static FEATURE fromValue(int v) {
076                        for (FEATURE f: FEATURE.values()) {
077                                if (f.value ==  v)
078                                        return f;
079                        }
080                        return null;
081                }
082
083        }
084
085        // Parse the features into FEATURE -> control code map
086        private static Map<FEATURE, Integer> tokenize(byte [] tlv) {
087                HashMap<FEATURE, Integer> m = new HashMap<FEATURE, Integer>();
088
089                if (tlv.length % 6 != 0) {
090                        throw new IllegalArgumentException("Bad response length: " + tlv.length);
091                }
092                for (int i = 0; i < tlv.length; i += 6) {
093                        int ft = tlv [i+0] & 0xFF;
094                        FEATURE f = FEATURE.fromValue(ft);
095                        byte[] c = Arrays.copyOfRange(tlv, i+2, i+6);
096                        ByteBuffer buffer = ByteBuffer.wrap(c);
097                        buffer.order(ByteOrder.BIG_ENDIAN);
098                        int ci = buffer.getInt();
099                        m.put(f, ci);
100                }
101                return m;
102        }
103
104        private  void parse_pin_properties(byte [] prop) {
105                if (prop.length == 4) {
106                        int cols = prop[0] & 0xFF;
107                        int rows = prop[1] & 0xFF;
108                        int pin = prop[2] & 0xFF;
109                        int timeout = prop[3] & 0xFF;
110                        if (rows > 0 && cols > 0 ) {
111                                display = true;
112                        }
113                        //                      System.out.println("COLS: " + cols);
114                        //                      System.out.println("ROWS: " + rows);
115                        //                      System.out.println("PIN : " + pin);
116                        //                      System.out.println("TIME: " + timeout);
117                } else {
118                        // XXX: older specification had size of 8
119                        throw new IllegalArgumentException("Bad PIN properties length: " + prop.length);
120                }
121        }

123        private static void parse_tlv_properties(byte [] tlv) {
124                for (int i = 0; i< tlv.length;) {
125                        int t = tlv[i+0] & 0xFF;
126                        int l = tlv[i+1] & 0xFF;
127                        byte [] v =  Arrays.copyOfRange(tlv, i + 2, i + 2 + l);
128                        i += v.length + 2;
129                        System.out.println(Integer.toHexString(t) + "=" + HexUtils.encodeHexString(v));
130                }
131        }
132        public static int CARD_CTL_CODE(int c) {
133                String os = System.getProperty("os.name", "unknown").toLowerCase();
134                if (os.indexOf("windows") != -1) {
135                        return 0x31 << 16  | c << 2;
136                } else {
137                        return 0x42000000 + c;
138                }
139        }
140
141        public void probe() throws CardException {
142                // probe() only makes sense when applied to terminal
143                if (t != null && c == null) {
144                        c = t.connect("DIRECT");
145                }
146                try {
147                        // Probe for features.
148                        byte [] resp = c.transmitControlCommand(CM_IOCTL_GET_FEATURE_REQUEST, new byte[]{});
149
150                        // Parse features
151                        Map<FEATURE, Integer> props = tokenize(resp);
152                        features.putAll(props);
153
154                        // Get PIN properties, if possible
155                        if (props.containsKey(FEATURE.IFD_PIN_PROPERTIES)) {
156                                resp = c.transmitControlCommand(props.get(FEATURE.IFD_PIN_PROPERTIES), new byte[]{});
157                                if (resp != null && resp.length > 0) {
158                                        parse_pin_properties(resp);
159                                }
160                        }
161
162                        // Get other properties
163                        if (props.containsKey(FEATURE.GET_TLV_PROPERTIES)) {
164                                resp = c.transmitControlCommand(props.get(FEATURE.GET_TLV_PROPERTIES), new byte[]{});
165                                if (resp != null && resp.length > 0) {
166                                        //parse_tlv_properties(resp);
167                                }
168                        }
169                } finally {
170                        c.disconnect(false);
171                }
172        }
173
174        public PinPadTerminal(CardTerminal terminal) {
175                t = terminal;
176        }
177
178        public PinPadTerminal(Card card) {
179                c = card;
180        }
181
182        public boolean canVerify() {
183                if (features.containsKey(FEATURE.VERIFY_PIN_DIRECT) || (features.containsKey(FEATURE.VERIFY_PIN_START) && features.containsKey(FEATURE.VERIFY_PIN_FINISH))) {
184                        return true;
185                }
186                return false;
187        }
188        public boolean canModify() {
189                if (features.containsKey(FEATURE.MODIFY_PIN_DIRECT) || (features.containsKey(FEATURE.MODIFY_PIN_START) && features.containsKey(FEATURE.MODIFY_PIN_FINISH))) {
190                        return true;
191                }
192                return false;
193        }
194        public boolean hasDisplay() {
195                return display;
196        }
197
198        @Override
199        public void close() throws IOException, CardException {
200                if (t != null && c != null) {
201                        c.disconnect(false); // FIXME: might be true
202                }
203        }
