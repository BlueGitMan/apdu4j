092        private static OptionSet parseOptions(String [] argv) throws IOException {
093                OptionSet args = null;
094                OptionParser parser = new OptionParser();
095                parser.acceptsAll(Arrays.asList("l", CMD_LIST), "list readers");
096                parser.acceptsAll(Arrays.asList("p", OPT_PROVIDER), "specify provider").withRequiredArg();
097                parser.acceptsAll(Arrays.asList("v", OPT_VERBOSE), "be verbose");
098                parser.acceptsAll(Arrays.asList("d", OPT_DEBUG), "enable debug");
099                parser.acceptsAll(Arrays.asList("e", OPT_ERROR), "fail if not 0x9000");
100                parser.acceptsAll(Arrays.asList("h", OPT_HELP), "show help");
101                parser.acceptsAll(Arrays.asList("r", OPT_READER), "use reader").withRequiredArg();
102                parser.acceptsAll(Arrays.asList("a", CMD_APDU), "send APDU").withRequiredArg();
103                parser.acceptsAll(Arrays.asList("w", OPT_WEB), "open ATR in web");
104                parser.accepts(OPT_DUMP, "save dump to file").withRequiredArg().ofType(File.class);
105                parser.accepts(OPT_REPLAY, "replay command from dump").withRequiredArg().ofType(File.class);
106
107                parser.accepts(OPT_SUN, "load SunPCSC");
108                parser.accepts(OPT_JNA, "load jnasmartcardio");
109                parser.accepts(OPT_CONNECT, "connect to host:port").withRequiredArg();
110                parser.accepts(OPT_PROVIDERS, "list providers");
111                parser.accepts(OPT_ALL, "process all readers");
112                parser.accepts(OPT_WAIT, "wait for card insertion");
113                parser.accepts(OPT_T0, "use T=0");
114                parser.accepts(OPT_T1, "use T=1");
115                parser.accepts(OPT_NO_GET_RESPONSE, "don't use GET RESPONSE with SunPCSC");
116                parser.accepts(OPT_LIB, "use specific PC/SC lib with SunPCSC").withRequiredArg();
117                parser.accepts(OPT_PROVIDER_TYPE, "provider type if not PC/SC").withRequiredArg();
118
119
120
121                // Parse arguments
122                try {
123                        args = parser.parse(argv);
124                        // Try to fetch all values so that format is checked before usage
125                        for (String s: parser.recognizedOptions().keySet()) {args.valuesOf(s);}
126                } catch (OptionException e) {
127                        if (e.getCause() != null) {
128                                System.err.println(e.getMessage() + ": " + e.getCause().getMessage());
129                        } else {
130                                System.err.println(e.getMessage());
131                        }
132                        System.err.println();
133                        help_and_exit(parser, System.err);
134                }
135                if (args.has(OPT_HELP)) {
136                        help_and_exit(parser, System.out);
137                }
138                return args;
139        }

141        public static void main(String[] argv) throws Exception {
142                OptionSet args = parseOptions(argv);
143                if (args.has(OPT_VERBOSE)) {
144                        verbose = true;
145                }
146                if (args.has(OPT_DEBUG)) {
147                        debug = true;
148                }
149
150                // List TerminalFactory providers
151                if (args.has(OPT_PROVIDERS)) {
152                        Provider providers[] = Security.getProviders("TerminalFactory.PC/SC");
153                        if (providers != null) {
154                                System.out.println("Existing TerminalFactory providers:");
155                                for (Provider p: providers) {
156                                        System.out.println(p.getName());
157                                }
158                        }
159                }
160
161                // Fix properties on non-windows platforms
162                TerminalManager.fixPlatformPaths();
163
164                // Only applies to SunPCSC
165                if (args.has(OPT_NO_GET_RESPONSE)) {
166                        System.setProperty("sun.security.smartcardio.t0GetResponse", "false");
167                        System.setProperty("sun.security.smartcardio.t1GetResponse", "false");
168                }
169
170                // Override PC/SC library path
171                if (args.has(OPT_LIB)) {
172                        System.setProperty("sun.security.smartcardio.library", (String) args.valueOf(OPT_LIB));
173                }
174
175                TerminalFactory tf = null;
176                CardTerminals terminals = null;
177
178                try {
179                        // Get a terminal factory
180                        if (args.has(OPT_PROVIDER)) {
181                                String pn = (String)args.valueOf(OPT_PROVIDER);
182                                String pt = (String)args.valueOf(OPT_PROVIDER_TYPE);
183                                tf = loadFactory(pn, pt);
184                        } else if (args.has(OPT_SUN)) {
185                                tf = loadFactory(SUN_CLASS, null);
186                        } else if (args.has(OPT_JNA)) {
187                                tf = loadFactory(JNA_CLASS, null);
188                        } else {
189                                tf = TerminalFactory.getDefault();
190                        }
191
192                        if (verbose) {
193                                System.out.println("# Using " + tf.getProvider().getClass().getCanonicalName() + " - " + tf.getProvider());
194                                if (System.getProperty(TerminalManager.lib_prop) != null) {
195                                        System.out.println("# " + TerminalManager.lib_prop + "=" + System.getProperty(TerminalManager.lib_prop));
196                                }
197                        }
198                        // Get all terminals
199                        terminals = tf.terminals();
200                } catch (Exception e) {
201                        // XXX: we catch generic Exception here to avoid importing JNA.
202                        // Try to get a meaningful message
203                        String msg = TerminalManager.getExceptionMessage(e);
204                        if (msg == null)
205                                msg = e.getMessage();
206                        System.out.println("No readers: " + msg);
207                        System.exit(1);
208                }
209
210                // Terminals to work on
211                List<CardTerminal> do_readers = new ArrayList<CardTerminal>();
212
213                try {
214                        // List Terminals
215                        if (args.has(CMD_LIST)) {
216                                List<CardTerminal> terms = terminals.list();
217                                if (verbose) {
218                                        System.out.println("# Found " + terms.size() + " terminal" + (terms.size() == 1 ? "" : "s"));
219                                }
220                                if (terms.size() == 0) {
221                                        System.err.println("No readers found");
222                                        System.exit(1);
223                                }
224                                for (CardTerminal t: terms) {
225                                        String vmd = " ";
226                                        try (PinPadTerminal pp = new PinPadTerminal(t)) {
227                                                pp.probe();
228                                                // Verify, Modify, Display
229                                                if (verbose) {
230                                                        vmd += "[";
231                                                        vmd += pp.canVerify() ? "V":" ";
232                                                        vmd += pp.canModify() ? "M":" ";
233                                                        vmd += pp.hasDisplay() ? "D":" ";
234                                                        vmd += "] ";
235                                                }
236                                        } catch (CardException e) {
237                                                if (verbose) {
238                                                        System.err.println("Could not probe PinPad: " + e.getMessage());
239                                                }
240                                        }
241
242
243                                        System.out.println((t.isCardPresent() ? "[*]" : "[ ]") + vmd + t.getName());
244
245                                        if (args.has(OPT_VERBOSE) && t.isCardPresent()) {
246                                                Card c = t.connect("DIRECT");
247                                                String atr = HexUtils.encodeHexString(c.getATR().getBytes()).toUpperCase();
248                                                c.disconnect(false);
249                                                System.out.println("          " + atr);
250                                                if (args.has(OPT_WEB)) {
251                                                        String url = "http://smartcard-atr.appspot.com/parse?ATR=" + atr;
252                                                        if (Desktop.isDesktopSupported()) {
253                                                                Desktop.getDesktop().browse(new URI(url + "&from=apdu4j"));
254                                                        } else {
255                                                                System.out.println("          " + url);
256                                                        }
257                                                }
258                                        }
259                                }
260                        }
261
262                        // Select terminals to work on
263                        if (args.has(OPT_READER)) {
264                                String reader = (String) args.valueOf(OPT_READER);
265                                CardTerminal t = terminals.getTerminal(reader);
266                                if (t == null) {
267                                        System.err.println("Reader \"" + reader + "\" not found.");
268                                        System.exit(1);
269                                }
270                                do_readers = Arrays.asList(t);
271                        } else {
272                                do_readers = terminals.list(State.CARD_PRESENT);
273                                if (do_readers.size() > 1 && !args.hasArgument(OPT_ALL)) {
274                                        System.err.println("More than one reader with a card found.");
275                                        System.err.println("Run with --"+OPT_ALL+" to work with all found cards");
276                                        System.exit(1);
277                                } else if (do_readers.size() == 0 && !args.has(CMD_LIST)) {
278                                        // But if there is a single reader, wait for a card insertion
279                                        List<CardTerminal> empty = terminals.list(State.CARD_ABSENT);
280                                        if (empty.size() == 1 && args.has(OPT_WAIT)) {
281                                                CardTerminal rdr = empty.get(0);
282                                                System.out.println("Please enter a card into " + rdr.getName());
283                                                if (!empty.get(0).waitForCardPresent(30000)) {
284                                                        System.out.println("Timeout.");
285                                                } else {
286                                                        do_readers = Arrays.asList(rdr);
287                                                }
288                                        } else {
289                                                System.err.println("No reader with a card found!");
290                                                System.exit(1);
291                                        }
292                                }
293                        }
294
295                } catch (CardException e) {
296                        System.out.println("Could not list readers: " + TerminalManager.getExceptionMessage(e));
297                        e.printStackTrace();
298                }
299
300                for (CardTerminal t: do_readers) {
301                        work(t, args);
302                }
303        }

305        private static void work(CardTerminal reader, OptionSet args) throws CardException {
306                if (!reader.isCardPresent()) {
307                        System.out.println("No card in " + reader.getName());
308                        return;
309                }
310
311                if (debug) {
312                        FileOutputStream o = null;
313                        if (args.has(OPT_DUMP)) {
314                                try {
315                                        o = new FileOutputStream((File) args.valueOf(OPT_DUMP));
316                                } catch (FileNotFoundException e) {
317                                        System.err.println("Can not dump to " + args.valueOf(OPT_DUMP));
318                                }
319                        }
320                        reader = LoggingCardTerminal.getInstance(reader, o);
321                }
322                // This allows to override the protocol for RemoteTerminal as well.
323                final String protocol;
324                if (args.has(OPT_T0)) {
325                        protocol = "T=0";
326                } else if (args.has(OPT_T1)) {
327                        protocol = "T=1";
328                } else {
329                        protocol = "*";
330                }
331                if (args.has(CMD_APDU))  {
332
333                        Card c = null;
334                        try {
335                                c = reader.connect(protocol);
336
337                                if (args.has(CMD_APDU)) {
338                                        for (Object s: args.valuesOf(CMD_APDU)) {
339                                                CommandAPDU a = new CommandAPDU(HexUtils.stringToBin((String)s));
340                                                ResponseAPDU r = c.getBasicChannel().transmit(a);
341                                                if (args.has(OPT_ERROR) && r.getSW() != 0x9000) {
342                                                        System.out.println("Card returned " + String.format("%04X", r.getSW()) + ", exiting!");
343                                                        return;
344                                                }
345                                        }
346                                }
347                        } catch (CardException e) {
348                                if (TerminalManager.getExceptionMessage(e) != null) {
349                                        System.out.println("PC/SC failure: " + TerminalManager.getExceptionMessage(e));
350                                } else {
351                                        throw e;
352                                }
353                        } finally {
354                                if (c != null) {
355                                        c.disconnect(true);
356                                }
357                        }
358                } else if (args.has(OPT_CONNECT)) {
359                        String[] hostport = ((String) args.valueOf(OPT_CONNECT)).split(":");
360                        // FIXME: move this check earlier
361                        if (hostport.length != 2) {
362                                throw new IllegalArgumentException("JSON target must be host:port pair!");
363                        }
364                        // Construct a remote terminal, by default ignoring certificate errors and using SocketTransport
365                        try (SocketTransport s = SocketTransport.connect(hostport[0], Integer.valueOf(hostport[1]), null)) {
366                                if (args.has(OPT_VERBOSE)) {
367                                        s.beVerbose(true);
368                                }
369                                // Connect the socket and the terminal
370                                CmdlineRemoteTerminal c = new CmdlineRemoteTerminal(s, reader);
371                                c.forceProtocol(protocol);
372                                // Run
373                                c.run();
374                        } catch (IOException e) {
375                                System.err.println("Communication error: " + e.getMessage());
376                        }
377                }
378        }

380        private static TerminalFactory loadFactory(String pn, String type) throws NoSuchAlgorithmException {
381                TerminalFactory tf = null;
382                try {
383                        Class<?> cls = Class.forName(pn);
384                        Provider p = (Provider) cls.getConstructor().newInstance();
385                        tf = TerminalFactory.getInstance(type == null ? "PC/SC" : type, null, p);
386                } catch (ClassNotFoundException | InstantiationException | IllegalAccessException | IllegalArgumentException | InvocationTargetException | NoSuchMethodException | SecurityException e) {
387                        throw new RuntimeException("Could not load " + pn, e);
388                } catch (NoSuchAlgorithmException e) {
389                        if (e.getCause() != null) {
390                                String causename = e.getCause().getClass().getCanonicalName();
391                                if (causename != null) {
392                                        if (causename.equalsIgnoreCase("java.lang.UnsupportedOperationException")) {
393                                                throw new NoSuchAlgorithmException(e.getCause().getMessage());
394                                        }
395                                        if (causename.equalsIgnoreCase("java.lang.UnsatisfiedLinkError")) {
396                                                throw new NoSuchAlgorithmException(e.getCause().getMessage());
397                                        }
398                                }
399                        }
400                        throw e;
401                }
402                return tf;
403        }

405        private static void help_and_exit(OptionParser parser, PrintStream o) throws IOException {
406                System.err.println("# apdu4j command line utility\n");
407                parser.printHelpOn(o);
408                System.exit(1);
409        }
