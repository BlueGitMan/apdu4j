107        private static OptionSet parseOptions(String [] argv) throws IOException {
108                OptionSet args = null;
109                OptionParser parser = new OptionParser();
110                parser.acceptsAll(Arrays.asList("l", CMD_LIST), "list readers");
111                parser.acceptsAll(Arrays.asList("p", OPT_PROVIDER), "specify provider").withRequiredArg();
112                parser.acceptsAll(Arrays.asList("v", OPT_VERBOSE), "be verbose");
113                parser.acceptsAll(Arrays.asList("e", OPT_ERROR), "fail if not 0x9000");
114                parser.acceptsAll(Arrays.asList("h", OPT_HELP), "show help");
115                parser.acceptsAll(Arrays.asList("r", OPT_READER), "use reader").withRequiredArg();
116                parser.acceptsAll(Arrays.asList("a", CMD_APDU), "send APDU").withRequiredArg();
117                parser.acceptsAll(Arrays.asList("w", OPT_WEB), "open ATR in web");
118                parser.accepts(OPT_VERSION, "show version information");
119                parser.accepts(OPT_DUMP, "save dump to file").withRequiredArg().ofType(File.class);
120                parser.accepts(OPT_REPLAY, "replay command from dump").withRequiredArg().ofType(File.class);
121
122                parser.accepts(OPT_SUN, "load SunPCSC");
123                parser.accepts(OPT_JNA, "load jnasmartcardio");
124                parser.accepts(OPT_CONNECT, "connect to host:port or URL").withRequiredArg();
125                parser.accepts(OPT_PINNED, "require certificate").withRequiredArg().ofType(File.class);
126
127                parser.accepts(OPT_PROVIDERS, "list providers");
128                parser.accepts(OPT_ALL, "process all readers");
129                parser.accepts(OPT_WAIT, "wait for card insertion");
130                parser.accepts(OPT_T0, "use T=0");
131                parser.accepts(OPT_T1, "use T=1");
132                parser.accepts(OPT_TEST_SERVER, "run a test server on port 10000");
133                parser.accepts(OPT_NO_GET_RESPONSE, "don't use GET RESPONSE with SunPCSC");
134                parser.accepts(OPT_LIB, "use specific PC/SC lib with SunPCSC").withRequiredArg();
135                parser.accepts(OPT_PROVIDER_TYPE, "provider type if not PC/SC").withRequiredArg();
136
137
138                // Parse arguments
139                try {
140                        args = parser.parse(argv);
141                        // Try to fetch all values so that format is checked before usage
142                        for (String s: parser.recognizedOptions().keySet()) {args.valuesOf(s);}
143                } catch (OptionException e) {
144                        if (e.getCause() != null) {
145                                System.err.println(e.getMessage() + ": " + e.getCause().getMessage());
146                        } else {
147                                System.err.println(e.getMessage());
148                        }
149                        System.err.println();
150                        help_and_exit(parser, System.err);
151                }
152                if (args.has(OPT_HELP)) {
153                        help_and_exit(parser, System.out);
154                }
155                return args;
156        }

158        public static void main(String[] argv) throws Exception {
159                OptionSet args = parseOptions(argv);
160
161                if (args.has(OPT_VERBOSE)) {
162                        verbose = true;
163                        // Set up slf4j simple in a way that pleases us
164                        System.setProperty("org.slf4j.simpleLogger.defaultLogLevel", "debug");
165                        System.setProperty("org.slf4j.simpleLogger.showThreadName", "true");
166                        System.setProperty("org.slf4j.simpleLogger.showShortLogName", "true");
167                        System.setProperty("org.slf4j.simpleLogger.levelInBrackets", "true");
168                } else {
169                        System.setProperty("org.slf4j.simpleLogger.defaultLogLevel", "warn");
170                }
171
172                if (args.has(OPT_VERSION)) {
173                        String version = getVersion(SCTool.class);
174                        // Append host information
175                        version += "\nRunning on " + System.getProperty("os.name");
176                        version += " " + System.getProperty("os.version");
177                        version += " " + System.getProperty("os.arch");
178                        version += ", Java " + System.getProperty("java.version");
179                        version += " by " + System.getProperty("java.vendor");
180                        System.out.println(version);
181                }
182                if (args.has(OPT_TEST_SERVER)) {
183                        // TODO: have the possibility to run SocketServer as well?
184                        RemoteTerminalServer srv = new RemoteTerminalServer(TestServer.class);
185                        srv.start();
186                        System.console().readLine("Press enter to stop\n");
187                        srv.stop(1);
188                        System.exit(0);
189                }
190
191                // List TerminalFactory providers
192                if (args.has(OPT_PROVIDERS)) {
193                        Provider providers[] = Security.getProviders("TerminalFactory.PC/SC");
194                        if (providers != null) {
195                                System.out.println("Existing TerminalFactory providers:");
196                                for (Provider p: providers) {
197                                        System.out.println(p.getName());
198                                }
199                        }
200                }
201
202                // Fix properties on non-windows platforms
203                TerminalManager.fixPlatformPaths();
204
205                // Only applies to SunPCSC
206                if (args.has(OPT_NO_GET_RESPONSE)) {
207                        System.setProperty("sun.security.smartcardio.t0GetResponse", "false");
208                        System.setProperty("sun.security.smartcardio.t1GetResponse", "false");
209                }
210
211                // Override PC/SC library path
212                if (args.has(OPT_LIB)) {
213                        System.setProperty("sun.security.smartcardio.library", (String) args.valueOf(OPT_LIB));
214                }
215
216                TerminalFactory tf = null;
217                CardTerminals terminals = null;
218
219                try {
220                        // Get a terminal factory
221                        if (args.has(OPT_PROVIDER)) {
222                                String pn = (String)args.valueOf(OPT_PROVIDER);
223                                String pt = (String)args.valueOf(OPT_PROVIDER_TYPE);
224                                tf = loadFactory(pn, pt);
225                        } else if (args.has(OPT_SUN)) {
226                                tf = loadFactory(SUN_CLASS, null);
227                        } else if (args.has(OPT_JNA)) {
228                                tf = loadFactory(JNA_CLASS, null);
229                        } else {
230                                tf = TerminalFactory.getDefault();
231                        }
232
233                        if (verbose) {
234                                System.out.println("# Using " + tf.getProvider().getClass().getCanonicalName() + " - " + tf.getProvider());
235                                if (System.getProperty(TerminalManager.lib_prop) != null) {
236                                        System.out.println("# " + TerminalManager.lib_prop + "=" + System.getProperty(TerminalManager.lib_prop));
237                                }
238                        }
239                        // Get all terminals
240                        terminals = tf.terminals();
241                } catch (Exception e) {
242                        // XXX: we catch generic Exception here to avoid importing JNA.
243                        // Try to get a meaningful message
244                        String msg = TerminalManager.getExceptionMessage(e);
245                        if (msg == null)
246                                msg = e.getMessage();
247                        System.out.println("No readers: " + msg);
248                        System.exit(1);
249                }
250
251                // Terminals to work on
252                List<CardTerminal> do_readers = new ArrayList<CardTerminal>();
253
254                try {
255                        // List Terminals
256                        if (args.has(CMD_LIST)) {
257                                List<CardTerminal> terms = terminals.list();
258                                if (verbose) {
259                                        System.out.println("# Found " + terms.size() + " terminal" + (terms.size() == 1 ? "" : "s"));
260                                }
261                                if (terms.size() == 0) {
262                                        System.err.println("No readers found");
263                                        System.exit(1);
264                                }
265                                for (CardTerminal t: terms) {
266                                        String vmd = " ";
267                                        try (PinPadTerminal pp = new PinPadTerminal(t)) {
268                                                pp.probe();
269                                                // Verify, Modify, Display
270                                                if (verbose) {
271                                                        vmd += "[";
272                                                        vmd += pp.canVerify() ? "V":" ";
273                                                        vmd += pp.canModify() ? "M":" ";
274                                                        vmd += pp.hasDisplay() ? "D":" ";
275                                                        vmd += "] ";
276                                                }
277                                        } catch (CardException e) {
278                                                if (verbose) {
279                                                        System.err.println("Could not probe PinPad: " + e.getMessage());
280                                                }
281                                        }
282
283
284                                        System.out.println((t.isCardPresent() ? "[*]" : "[ ]") + vmd + t.getName());
285
286                                        if (args.has(OPT_VERBOSE) && t.isCardPresent()) {
287                                                Card c = t.connect("DIRECT");
288                                                String atr = HexUtils.encodeHexString(c.getATR().getBytes()).toUpperCase();
289                                                c.disconnect(false);
290                                                System.out.println("          " + atr);
291                                                if (args.has(OPT_WEB)) {
292                                                        String url = "http://smartcard-atr.appspot.com/parse?ATR=" + atr;
293                                                        if (Desktop.isDesktopSupported()) {
294                                                                Desktop.getDesktop().browse(new URI(url + "&from=apdu4j"));
295                                                        } else {
296                                                                System.out.println("          " + url);
297                                                        }
298                                                }
299                                        }
300                                }
301                        }
302
303                        // Select terminals to work on
304                        if (args.has(OPT_READER)) {
305                                String reader = (String) args.valueOf(OPT_READER);
306                                CardTerminal t = terminals.getTerminal(reader);
307                                if (t == null) {
308                                        System.err.println("Reader \"" + reader + "\" not found.");
309                                        System.exit(1);
310                                }
311                                do_readers = Arrays.asList(t);
312                        } else {
313                                do_readers = terminals.list(State.CARD_PRESENT);
314                                if (do_readers.size() > 1 && !args.hasArgument(OPT_ALL)) {
315                                        System.err.println("More than one reader with a card found.");
316                                        System.err.println("Run with --"+OPT_ALL+" to work with all found cards");
317                                        System.exit(1);
318                                } else if (do_readers.size() == 0 && !args.has(CMD_LIST)) {
319                                        // But if there is a single reader, wait for a card insertion
320                                        List<CardTerminal> empty = terminals.list(State.CARD_ABSENT);
321                                        if (empty.size() == 1 && args.has(OPT_WAIT)) {
322                                                CardTerminal rdr = empty.get(0);
323                                                System.out.println("Please enter a card into " + rdr.getName());
324                                                if (!empty.get(0).waitForCardPresent(30000)) {
325                                                        System.out.println("Timeout.");
326                                                } else {
327                                                        do_readers = Arrays.asList(rdr);
328                                                }
329                                        } else {
330                                                System.err.println("No reader with a card found!");
331                                                System.exit(1);
332                                        }
333                                }
334                        }
335
336                } catch (CardException e) {
337                        System.out.println("Could not list readers: " + TerminalManager.getExceptionMessage(e));
338                        e.printStackTrace();
339                }
340
341                for (CardTerminal t: do_readers) {
342                        work(t, args);
343                }
344        }

346        private static void work(CardTerminal reader, OptionSet args) throws CardException {
347                if (!reader.isCardPresent()) {
348                        System.out.println("No card in " + reader.getName());
349                        return;
350                }
351
352                FileOutputStream o = null;
353                if (args.has(OPT_DUMP)) {
354                        try {
355                                o = new FileOutputStream((File) args.valueOf(OPT_DUMP));
356                        } catch (FileNotFoundException e) {
357                                System.err.println("Can not dump to " + args.valueOf(OPT_DUMP));
358                        }
359                }
360                reader = LoggingCardTerminal.getInstance(reader, o);
361                // This allows to override the protocol for RemoteTerminal as well.
362                final String protocol;
363                if (args.has(OPT_T0)) {
364                        protocol = "T=0";
365                } else if (args.has(OPT_T1)) {
366                        protocol = "T=1";
367                } else {
368                        protocol = "*";
369                }
370                if (args.has(CMD_APDU))  {
371
372                        Card c = null;
373                        try {
374                                c = reader.connect(protocol);
375
376                                if (args.has(CMD_APDU)) {
377                                        for (Object s: args.valuesOf(CMD_APDU)) {
378                                                CommandAPDU a = new CommandAPDU(HexUtils.stringToBin((String)s));
379                                                ResponseAPDU r = c.getBasicChannel().transmit(a);
380                                                if (args.has(OPT_ERROR) && r.getSW() != 0x9000) {
381                                                        System.out.println("Card returned " + String.format("%04X", r.getSW()) + ", exiting!");
382                                                        return;
383                                                }
384                                        }
385                                }
386                        } catch (CardException e) {
387                                if (TerminalManager.getExceptionMessage(e) != null) {
388                                        System.out.println("PC/SC failure: " + TerminalManager.getExceptionMessage(e));
389                                } else {
390                                        throw e;
391                                }
392                        } finally {
393                                if (c != null) {
394                                        c.disconnect(true);
395                                }
396                        }
397                } else if (args.has(OPT_CONNECT)) {
398                        String remote = (String) args.valueOf(OPT_CONNECT);
399                        JSONMessagePipe transport = null;
400
401                        try {
402                                if (remote.startsWith("http://") || remote.startsWith("https://")) {
403                                        if (args.has(OPT_PINNED)) {
404                                                transport = HTTPTransport.open(new URL(remote), certFromPEM(((File)args.valueOf(OPT_PINNED)).getPath()));
405                                        } else {
406                                                transport = HTTPTransport.open(new URL(remote), null);
407                                        }
408                                } else {
409                                        String[] hostport = remote.split(":");
410                                        // FIXME: move this check earlier
411                                        if (hostport.length != 2) {
412                                                throw new IllegalArgumentException("JSON target must be host:port pair!");
413                                        }
414                                        transport = SocketTransport.connect(hostport[0], Integer.valueOf(hostport[1]), null);
415                                }
416
417                                // Connect the transport and the terminal
418                                CmdlineRemoteTerminal c = new CmdlineRemoteTerminal(transport, reader);
419                                c.forceProtocol(protocol);
420                                // Run
421                                c.run();
422                        } catch (IOException e) {
423                                System.err.println("Communication error: " + e.getMessage());
424                        } finally {
425                                if (transport != null)
426                                        transport.close();
427                        }
428                }
429        }

432        private static TerminalFactory loadFactory(String pn, String type) throws NoSuchAlgorithmException {
433                TerminalFactory tf = null;
434                try {
435                        Class<?> cls = Class.forName(pn);
436                        Provider p = (Provider) cls.getConstructor().newInstance();
437                        tf = TerminalFactory.getInstance(type == null ? "PC/SC" : type, null, p);
438                } catch (ClassNotFoundException | InstantiationException | IllegalAccessException | IllegalArgumentException | InvocationTargetException | NoSuchMethodException | SecurityException e) {
439                        throw new RuntimeException("Could not load " + pn, e);
440                } catch (NoSuchAlgorithmException e) {
441                        if (e.getCause() != null) {
442                                Class<?> cause = e.getCause().getClass();
443                                if (cause.equals(java.lang.UnsupportedOperationException.class)) {
444                                        throw new NoSuchAlgorithmException(e.getCause().getMessage());
445                                }
446                                if (cause.equals(java.lang.UnsatisfiedLinkError.class)) {
447                                        throw new NoSuchAlgorithmException(e.getCause().getMessage());
448                                }
449                        }
450                        throw e;
451                }
452                return tf;
453        }

461        private static X509Certificate certFromPEM(String path) throws IOException {
462                try {
463                        String pem =  new String(Files.readAllBytes(Paths.get(path)));
464                        String [] lines = pem.split("\n");
465                        String [] b64 = Arrays.copyOfRange(lines, 1, lines.length-1);
466                        byte [] bytes = Base64.getDecoder().decode(String.join("", b64));
467                        CertificateFactory cf;
468                        cf = CertificateFactory.getInstance("X.509");
469                        X509Certificate cert = (X509Certificate)cf.generateCertificate(new ByteArrayInputStream(bytes));
470                        return cert;
471                }
472                catch (CertificateException e) {
473                        throw new IOException(e);
474                }
475        }

477        public static String getVersion(Class<?> clazz) {
478                String version = "unknown-development";
479                try (InputStream versionfile = clazz.getResourceAsStream("/version.txt")) {
480                        if (versionfile != null) {
481                                BufferedReader vinfo = new BufferedReader(new InputStreamReader(versionfile));
482                                version = vinfo.readLine();
483                        }
484                } catch (IOException e) {
485                        version = "unknown-error";
486                }
487                return version;
488        }
