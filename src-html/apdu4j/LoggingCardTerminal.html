022package apdu4j;
023
024import java.io.OutputStream;
025import java.io.PrintStream;
026import java.nio.ByteBuffer;
027import java.text.SimpleDateFormat;
028import java.util.Arrays;
029import java.util.Calendar;
030
031import javax.smartcardio.ATR;
032import javax.smartcardio.Card;
033import javax.smartcardio.CardChannel;
034import javax.smartcardio.CardException;
035import javax.smartcardio.CardTerminal;
036import javax.smartcardio.CommandAPDU;
037import javax.smartcardio.ResponseAPDU;
038
039/**
040 * Log everything going through this CardTerminal, also hypothetical SCard API calls for connect/disconnect.
041 *
042 * Dump APDU-s to a file for automatic playback and inspection
043 *
044 * @author Martin Paljak
045 */
046public class LoggingCardTerminal extends CardTerminal {
047
048        // The actual terminal
049        protected CardTerminal terminal = null;
050        protected PrintStream dump = null;
051        protected PrintStream log = null;
052
053        public static LoggingCardTerminal getInstance(CardTerminal term) {
054                return new LoggingCardTerminal(term, System.out, null);
055        }
056
057        public static LoggingCardTerminal getInstance(CardTerminal term, OutputStream dump) {
058                return new LoggingCardTerminal(term, System.out, dump);
059        }
060
061        private LoggingCardTerminal (CardTerminal term, OutputStream log, OutputStream dump) {
062                this.terminal = term;
063                this.log = new PrintStream(log, true);
064                if (dump != null) {
065                        this.dump = new PrintStream(dump, true);
066                }
067        }
068
069        @Override
070        public Card connect(String arg0) throws CardException {
071                return new LoggingCard(terminal, arg0);
072        }
073
074        @Override
075        public String getName() {
076                return terminal.getName();
077        }
078
079        @Override
080        public boolean isCardPresent() throws CardException {
081                return terminal.isCardPresent();
082        }
083
084        @Override
085        public boolean waitForCardAbsent(long arg0) throws CardException {
086                return terminal.waitForCardAbsent(arg0);
087
088        }
089
090        @Override
091        public boolean waitForCardPresent(long arg0) throws CardException {
092                return terminal.waitForCardPresent(arg0);
093        }
094
095
096        public final class LoggingCard extends Card {
097                private final Card card;
098                private LoggingCard(CardTerminal term, String protocol) throws CardException {
099                        log.print("SCardConnect(\"" + terminal.getName() + "\", " + (protocol.equals("*") ? "T=*" : protocol) + ")");
100                        log.flush();
101                        card = terminal.connect(protocol);
102                        String atr = HexUtils.encodeHexString(card.getATR().getBytes());
103                        log.println(" -> " + card.getProtocol() + ", " + atr);
104                        if (dump != null) {
105                                String ts = new SimpleDateFormat("EEE, d MMM yyyy HH:mm:ss Z").format(Calendar.getInstance().getTime());
106                                dump.println("# Generated on " + ts + " by apdu4j");
107                                dump.println("# Using " + terminal.getName());
108                                dump.println("# ATR: " + atr);
109                                dump.println("# PROTOCOL: " + card.getProtocol());
110                                dump.println("#");
111                        }
112                }
113
114                @Override
115                public void beginExclusive() throws CardException {
116                        log.println("SCardBeginTransaction(\""+terminal.getName() +"\")");
117                        card.beginExclusive();
118                }
119
120                @Override
121                public void disconnect(boolean arg0) throws CardException {
122                        log.println("SCardDisconnect(\""+terminal.getName() +"\", " + arg0 +")");
123                        if (dump != null) {
124                                dump.close();
125                        }
126                        card.disconnect(arg0);
127                }
128
129                @Override
130                public void endExclusive() throws CardException {
131                        log.println("SCardEndTransaction()");
132                        card.endExclusive();
133                }
134
135                @Override
136                public ATR getATR() {
137                        return card.getATR();
138                }
139
140                @Override
141                public CardChannel getBasicChannel() {
142                        return new LoggingCardChannel(card);
143                }
144
145                @Override
146                public String getProtocol() {
147                        return card.getProtocol();
148                }
149
150                @Override
151                public CardChannel openLogicalChannel() throws CardException {
152                        throw new CardException("Logical channels not supported");
153                }
154
155                @Override
156                public byte[] transmitControlCommand(int arg0, byte[] arg1) throws CardException {
157                        throw new CardException("Control commands not (yet) supported");
158                }
159
160                public final class LoggingCardChannel extends CardChannel {
161                        private final CardChannel channel;
162                        private final Card card;
163                        public LoggingCardChannel(Card card) {
164                                this.card = card;
165                                this.channel = card.getBasicChannel();
166                        }
167                        @Override
168                        public void close() throws CardException {
169                                channel.close();
170                        }
171
172                        @Override
173                        public Card getCard() {
174                                return card;
175                        }
176
177                        @Override
178                        public int getChannelNumber() {
179                                return channel.getChannelNumber();
180                        }
181
182                        @Override
183                        public ResponseAPDU transmit(CommandAPDU apdu) throws CardException {
184                                byte [] cb = apdu.getBytes();
185                                int len_end = apdu.getData().length > 255 ? 7 : 5;
186                                log.print("A>> " + card.getProtocol() + " (4+" + String.format("%04d", apdu.getData().length) + ")");
187                                log.print(" " + HexUtils.encodeHexString(Arrays.copyOfRange(cb, 0, 4)));
188
189                                // Only if Case 2, 3 or 4 APDU
190                                if (cb.length > 4) {
191                                        int cmdlen = cb[ISO7816.OFFSET_LC] & 0xFF;
192                                        if (cmdlen == 0 && apdu.getData().length > 6) {
193                                                cmdlen = ((cb[ISO7816.OFFSET_LC +1] & 0xff << 8) | cb[ISO7816.OFFSET_LC+2] & 0xff);
194                                        }
195                                        log.print(" " + HexUtils.encodeHexString(Arrays.copyOfRange(cb, 4, len_end)));
196                                        log.print(" " + HexUtils.encodeHexString(Arrays.copyOfRange(cb, len_end, len_end + cmdlen)));
197                                        if (len_end + cmdlen < cb.length) {
198                                                log.println(" " + HexUtils.encodeHexString(Arrays.copyOfRange(cb, len_end + cmdlen, cb.length)));
199                                        } else {
200                                                log.println();
201                                        }
202                                } else {
203                                        log.println();
204                                }
205                                log.flush();
206
207                                long t = System.currentTimeMillis();
208                                ResponseAPDU response = channel.transmit(apdu);
209                                long ms = System.currentTimeMillis() - t;
210                                String time = ms + "ms";
211                                if (ms > 1000) {
212                                        time = ms / 1000 + "s" + ms % 1000 + "ms";
213                                }
214                                byte [] rb = response.getBytes();
215                                System.out.print("A<< (" + String.format("%04d", response.getData().length) + "+2) (" + time + ")");
216                                if (rb.length > 2) {
217                                        log.print(" " + HexUtils.encodeHexString(Arrays.copyOfRange(rb, 0, rb.length-2)));
218                                }
219                                log.println(" " + HexUtils.encodeHexString(Arrays.copyOfRange(rb, rb.length-2, rb.length)));
220                                if (dump != null) {
221                                        dump.println("# Sent\n" + HexUtils.encodeHexString(cb));
222                                        dump.println("# Received in " + time + "\n" + HexUtils.encodeHexString(rb));
223                                }
224                                return response;
225                        }
226
227                        @Override
228                        public int transmit(ByteBuffer cmd, ByteBuffer rsp) throws CardException {
229                                byte[] commandBytes = new byte[cmd.remaining()];
230                                cmd.get(commandBytes);
231                                cmd.position(0);
232
233                                log.println("B>> " + card.getProtocol() + " (" + commandBytes.length + ") " + HexUtils.encodeHexString(commandBytes));
234                                int response = channel.transmit(cmd, rsp);
235                                byte[] responseBytes = new byte[response];
236                                rsp.get(responseBytes);
237                                rsp.position(0);
238                                log.println("B<< (" + responseBytes.length + ") " + HexUtils.encodeHexString(responseBytes));
239                                if (dump != null) {
240                                        dump.println("# Sent\n" + HexUtils.encodeHexString(commandBytes));
241                                        dump.println("# Received\n" + HexUtils.encodeHexString(responseBytes));
242                                }
243                                return response;
244                        }
245                }
246        }
