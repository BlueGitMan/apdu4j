022package apdu4j.remote;
023
024import java.io.Console;
025import java.io.IOException;
026import java.nio.charset.StandardCharsets;
027import java.util.Map;
028
029import javax.smartcardio.CardException;
030import javax.smartcardio.CardTerminal;
031import javax.smartcardio.CommandAPDU;
032import javax.smartcardio.ResponseAPDU;
033
034import apdu4j.HexUtils;
035
036/**
037 * Client side implementation of the remote EMV terminal protocol that works
038 * from the command line.
039 *
040 * @author Martin Paljak
041 */
042public class CmdlineRemoteTerminal implements Runnable {
043        private JSONMessagePipe pipe;
044        // The terminal that is tunneled
045        private JSONCardTerminalClient jsonterminal;
046
047        public CmdlineRemoteTerminal(JSONMessagePipe pipe, CardTerminal terminal) {
048                this.pipe = pipe;
049                this.jsonterminal = new JSONCardTerminalClient(terminal, pipe);
050        }
051
052        @Override
053        public void run() {
054                try {
055                        // Initiate communication.
056                        pipe.send(JSONProtocol.cmd("start"));
057                        while (true) {
058                                // Read a message
059                                Map<String, Object> m = pipe.recv();
060                                // First try the terminal
061                                if (!jsonterminal.processMessage(m)) {
062                                        if (!processMessage(m))
063                                                break;
064                                }
065                        }
066                } catch (IOException e) {
067                        System.out.println("Messaging failed: " + e.getMessage());
068                }
069        }
070
071        public void forceProtocol(String protocol) {
072                jsonterminal.forceProtocol(protocol);
073        }
074        private void verify(Map<String, Object> msg) throws IOException {
075                if (!jsonterminal.isConnected()) {
076                        throw new IllegalStateException("Can not verify PIN codes if no connection to a card is established!");
077                }
078                Console c = System.console();
079                int p2 = ((Long)msg.get("p2")).intValue();
080                System.out.println((String)msg.get("text"));
081                byte[] pin = new String(c.readPassword("Enter PIN: ")).getBytes(StandardCharsets.UTF_8);
082                CommandAPDU verify = new CommandAPDU(0x00, 0x20, 0x00, p2, pin);
083                try {
084                        ResponseAPDU r = jsonterminal.card.getBasicChannel().transmit(verify);
085                        Map< String, Object> m = null;
086                        if (r.getSW() == 0x9000) {
087                                m = JSONProtocol.ok(msg);
088                        } else {
089                                m = JSONProtocol.nok(msg, "Verification failed");
090                                m.put("bytes", HexUtils.encodeHexString(r.getBytes()));
091                        }
092                        pipe.send(m);
093                } catch (CardException e) {
094                        pipe.send(JSONProtocol.nok(msg, e.getMessage()));
095                }
096        }
097
098        private void dialog(Map<String, Object> msg) throws IOException {
099                System.out.println("# " + msg.get("text"));
100                Map< String, Object> m = JSONProtocol.ok(msg);
101                boolean yes = false;
102
103                // lanterna requires work as it screws up Console.readPassword()
104                yes = get_yes_or_no_console();
105
106                if (!yes) {
107                        m.put("button", "red");
108                } else {
109                        m.put("button", "green");
110                }
111                pipe.send(m);
112        }
113
114        private void message(Map<String, Object> msg) throws IOException {
115                System.out.println("# " + (String)msg.get("text"));
116                pipe.send(JSONProtocol.ok(msg));
117        }
118        private boolean processMessage (Map<String, Object> msg) throws IOException {
119                if (!msg.containsKey("cmd")) {
120                        throw new IOException("No command in message: " + msg);
121                }
122                String cmd = (String) msg.get("cmd");
123                if (cmd.equals("MESSAGE")) {
124                        message(msg);
125                } else if (cmd.equals("VERIFY")) {
126                        verify(msg);
127                } else if (cmd.equals("DIALOG")) {
128                        dialog(msg);
129                } else if (cmd.equals("STOP")) {
130                        System.out.println("# Connection closed.");
131                        return false;
132                } else {
133                        System.err.println("No idea how to process: " + msg.toString());
134                        return false;
135                }
136                return true;
137        }
138
139        private boolean get_yes_or_no_console() {
140                Console c = System.console();
141                while (true) {
142                        String response = c.readLine("y/n ? ").trim();
143                        if (response.equalsIgnoreCase("y")) {
144                                return true;
145                        } else if (response.equalsIgnoreCase("n")) {
146                                return false;
147                        } else {
148                                System.out.println("Please enter 'y' or 'n' followed by ENTER");
149                        }
150                }
151        }
152
153}
