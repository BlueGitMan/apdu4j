022package apdu4j.remote;
023
024import java.io.Console;
025import java.io.IOException;
026import java.nio.charset.StandardCharsets;
027import java.util.Map;
028
029import javax.smartcardio.CardException;
030import javax.smartcardio.CardTerminal;
031import javax.smartcardio.CommandAPDU;
032import javax.smartcardio.ResponseAPDU;
033
034import com.googlecode.lanterna.input.KeyStroke;
035import com.googlecode.lanterna.input.KeyType;
036import com.googlecode.lanterna.terminal.DefaultTerminalFactory;
037import com.googlecode.lanterna.terminal.Terminal;
038
039import apdu4j.HexUtils;
040
041/**
042 * Client side implementation of the remote EMV terminal protocol that works
043 * from the command line.
044 *
045 * @author Martin Paljak
046 */
047public class CmdlineRemoteTerminal implements Runnable {
048        private JSONMessagePipe pipe;
049        // The terminal that is tunneled
050        private JSONCardTerminalClient jsonterminal;
051
052        public CmdlineRemoteTerminal(JSONMessagePipe pipe, CardTerminal terminal) {
053                this.pipe = pipe;
054                this.jsonterminal = new JSONCardTerminalClient(terminal, pipe);
055        }
056
057        @Override
058        public void run() {
059                while (true) {
060                        try {
061                                // Read a message
062                                Map<String, Object> m = pipe.recv();
063                                // First try the terminal
064                                if (!jsonterminal.processMessage(m)) {
065                                        processMessage(m);
066                                }
067                        } catch (IOException e) {
068                                System.out.println("Messaging failed: " + e.getMessage());
069                                break;
070                        }
071                }
072        }
073
074        public void forceProtocol(String protocol) {
075                jsonterminal.forceProtocol(protocol);
076        }
077        private void verify(Map<String, Object> msg) throws IOException {
078                if (!jsonterminal.isConnected()) {
079                        throw new IllegalStateException("Can not verify PIN codes if no connection to a card is established!");
080                }
081                Console c = System.console();
082                int p2 = ((Long)msg.get("p2")).intValue();
083                System.out.println((String)msg.get("text"));
084                byte[] pin = new String(c.readPassword("Enter PIN: ")).getBytes(StandardCharsets.UTF_8);
085                CommandAPDU verify = new CommandAPDU(0x00, 0x20, 0x00, p2, pin);
086                try {
087                        ResponseAPDU r = jsonterminal.card.getBasicChannel().transmit(verify);
088                        Map< String, Object> m = null;
089                        if (r.getSW() == 0x9000) {
090                                m = JSONProtocol.ok(msg);
091                        } else {
092                                m = JSONProtocol.nok(msg, "Verification failed");
093                                m.put("bytes", HexUtils.encodeHexString(r.getBytes()));
094                        }
095                        pipe.send(m);
096                } catch (CardException e) {
097                        pipe.send(JSONProtocol.nok(msg, e.getMessage()));
098                }
099        }
100
101        private void dialog(Map<String, Object> msg) throws IOException {
102                System.out.println("# " + msg.get("text"));
103                Map< String, Object> m = JSONProtocol.ok(msg);
104                boolean yes = false;
105
106                if (System.getProperty("os.name").equalsIgnoreCase("Linux")) {
107                        yes = get_yes_or_no_lantern();
108                } else {
109                        yes = get_yes_or_no_console();
110                }
111
112                if (!yes) {
113                        m.put("button", "red");
114                } else {
115                        m.put("button", "green");
116                }
117                pipe.send(m);
118        }
119
120        private void message(Map<String, Object> msg) throws IOException {
121                System.out.println("# " + (String)msg.get("text"));
122                pipe.send(JSONProtocol.ok(msg));
123        }
124        private void processMessage (Map<String, Object> msg) throws IOException {
125                if (!msg.containsKey("cmd")) {
126                        throw new IOException("No command in message: " + msg);
127                }
128                String cmd = (String) msg.get("cmd");
129                if (cmd.equals("MESSAGE")) {
130                        message(msg);
131                } else if (cmd.equals("VERIFY")) {
132                        verify(msg);
133                } else if (cmd.equals("DIALOG")) {
134                        dialog(msg);
135                } else if (cmd.equals("STOP")) {
136                        System.out.println("Disconnecting, STOP received.");
137                } else {
138                        System.out.println("No idea how to process: " + msg.toString());
139                }
140        }
141
142        private boolean get_yes_or_no_lantern() throws IOException {
143                Terminal term = null;
144                try {
145                        // Handle all the nice things.
146                        DefaultTerminalFactory tf = new DefaultTerminalFactory();
147                        // Otherwise a window is created on Linux
148                        tf.setSuppressSwingTerminalFrame(true);
149                        term = tf.createTerminal();
150                }
151                catch (IOException e) {
152                        // TODO Auto-generated catch block
153                        e.printStackTrace();
154                }
155                System.out.print("Y/Enter N/Escape ? ");
156                while (true) {
157                        KeyStroke k = term.readInput();
158                        if (k.getKeyType().equals(KeyType.Enter)) {
159                                return true;
160                        } else if (k.getKeyType().equals(KeyType.Escape)) {
161                                return false;
162                        } else if (k.getKeyType().equals(KeyType.Character)) {
163                                char c = k.getCharacter();
164                                if (c == 'y' || c == 'Y') {
165                                        return true;
166                                } else if (c == 'n' || c == 'N') {
167                                        return false;
168                                } else {
169                                        System.out.println("Please enter 'y' or 'n', enter, cancel");
170                                }
171                        } else {
172                                System.out.println("y/n ? ");
173                                System.out.println(k);
174                        }
175                }
176        }
177
178        private boolean get_yes_or_no_console() {
179                Console c = System.console();
180                while (true) {
181                        String response = c.readLine("y/n ? ").trim();
182                        if (response.equalsIgnoreCase("y")) {
183                                return true;
184                        } else if (response.equalsIgnoreCase("n")) {
185                                return false;
186                        } else {
187                                System.out.println("Please enter 'y' or 'n' followed by ENTER");
188                        }
189                }
190        }
191
192}
