022package apdu4j.remote;
023
024import java.io.IOException;
025import java.io.InputStream;
026import java.io.OutputStream;
027import java.net.HttpURLConnection;
028import java.net.URL;
029import java.nio.charset.Charset;
030import java.security.cert.X509Certificate;
031import java.util.HashMap;
032import java.util.Map;
033
034import javax.net.ssl.HttpsURLConnection;
035import javax.net.ssl.SSLSocketFactory;
036
037import org.json.simple.JSONObject;
038import org.json.simple.JSONValue;
039import org.slf4j.Logger;
040import org.slf4j.LoggerFactory;
041
042// Used by the client.
043public class HTTPTransport implements JSONMessagePipe {
044        private static Logger logger = LoggerFactory.getLogger(HTTPTransport.class);
045
046        private final URL u;
047        HttpURLConnection c = null;
048        private SSLSocketFactory f = null;
049
050        private HTTPTransport(URL url) {
051                this.u = url;
052        }
053
054
055
056        // Open a connection to a URL, with a pinned cert (or no checking)
057        public static HTTPTransport open(URL u, X509Certificate pinned) throws IOException {
058                HTTPTransport t = new HTTPTransport(u);
059                t.f = SocketTransport.get_ssl_socket_factory(pinned);
060                return t;
061        }
062
063
064        private HttpURLConnection connect() throws IOException {
065                c = (HttpURLConnection)u.openConnection();
066                c.setDoOutput(true);
067                c.setDoInput(true);
068                c.setRequestProperty("Content-Type", "application/json");
069                c.setRequestProperty("Accept", "application/json");
070                c.setRequestMethod("POST");
071
072                // Set SSL options
073                if (c instanceof HttpsURLConnection) {
074                        if (f != null) {
075                                HttpsURLConnection https = (HttpsURLConnection) c;
076                                https.setSSLSocketFactory(f);
077                        }
078                }
079                return c;
080        }
081
082        @Override
083        public synchronized void send(Map<String, Object> msg) throws IOException {
084                // Always open a new connection when sending.
085                c = connect();
086
087                JSONObject obj = new JSONObject(msg);
088                logger.trace("SEND: {}", obj.toJSONString());
089                byte[] data = obj.toJSONString().getBytes(Charset.forName("UTF-8"));
090                // Close the stream
091                try (OutputStream out = c.getOutputStream()) {
092                        out.write(data);
093                }
094        }
095
096        @Override
097        public synchronized Map<String, Object> recv() throws IOException {
098                HashMap<String, Object> r = new HashMap<>();
099                if (c.getResponseCode() == 200) {
100                        try (InputStream in = c.getInputStream()) {
101                                byte [] response = new byte[c.getContentLength()];
102                                in.read(response);
103                                JSONObject obj = (JSONObject) JSONValue.parse(new String(response, "UTF-8"));
104                                logger.trace("RECV: {}", obj.toJSONString());
105                                r.putAll(obj);
106                        }
107                } else {
108                        logger.trace("Got response code {}", c.getResponseCode());
109                        throw new IOException("Got response code " + c.getResponseCode());
110                }
111                return r;
112        }
113
114        @Override
115        public void close() {
116                // Disconnect, if connection open.
117                if (c!= null) {
118                        c.disconnect();
119                }
120        }
121}
