055/**
056 * Based on the <a href=
057 * "https://developer.chrome.com/extensions/nativeMessaging#native-messaging-host-protocol">
058 * NativeMessaging protocol from Chrome</a> (but uses big endian uint32
059 * as is common in network protocols). Messages bigger than 1024 bytes are
060 * rejected by this implementation and the underlying socket is closed.
061 *
062 * @author Martin Paljak
063 */
064public class SocketTransport implements JSONMessagePipe {
065        private static Logger logger = LoggerFactory.getLogger(SocketTransport.class);
066
067        private final Socket socket;
068        private final ByteBuffer length = ByteBuffer.allocate(Integer.SIZE / Byte.SIZE);
069
070        public SocketTransport(Socket s) {
071                socket = s;
072                length.order(ByteOrder.BIG_ENDIAN);
073        }
074
075
076        /**
077         * Connects to the mentioned host and port with SSL without checking certificate chain.
078         *
079         * @param host host to connect to
080         * @param port port to use
081         * @return instance of this class
082         * @throws IOException if establishing the connection fails
083         */
084        public static SocketTransport connect_insecure(String host, int port) throws IOException {
085                return connect(host, port, null);
086        }
087
088        // Returns a SSLSocketFactory that either does no checking or checks for a pinnned certificate
089        protected static SSLSocketFactory get_ssl_socket_factory(X509Certificate pinnedcert) throws IOException {
090                try {
091                        // Create a trust manager that does not validate certificate chains
092                        TrustManager[] trustAllCerts = new TrustManager[] {
093                                        new X509TrustManager() {
094                                                @Override
095                                                public X509Certificate[] getAcceptedIssuers() {
096                                                        return new X509Certificate[0];
097                                                }
098                                                @Override
099                                                public void checkServerTrusted(X509Certificate[] arg0, String arg1) throws CertificateException {
100                                                }
101                                                @Override
102                                                public void checkClientTrusted(X509Certificate[] arg0, String arg1) throws CertificateException {
103                                                }
104                                        }
105                        };
106
107                        // Trust managers for SSL context
108                        SSLContext sc = SSLContext.getInstance("TLS");
109
110                        if (pinnedcert != null) {
111                                TrustManagerFactory tmf = TrustManagerFactory.getInstance("X509");
112                                KeyStore ks = null;
113                                ks = KeyStore.getInstance(KeyStore.getDefaultType());
114                                // Generate an empty one
115                                ks.load(null, null);
116                                ks.setCertificateEntry("pinned", pinnedcert);
117                                tmf.init(ks);
118                                sc.init(null, tmf.getTrustManagers(), new java.security.SecureRandom());
119                        } else {
120                                sc.init(null, trustAllCerts, new java.security.SecureRandom());
121                        }
122                        // Connect with created parameters
123                        return sc.getSocketFactory();
124                } catch (KeyManagementException | KeyStoreException | NoSuchAlgorithmException | CertificateException e) {
125                        throw new IOException("Could not connect", e);
126                }
127        }

129        /**
130         * Connects to the mentioned host and port with SSL and checks the certificate against the pinned version.
131         *
132         * @param host host to connect to
133         * @param port port to use
134         * @param pinnedcert expected certificate of the server
135         * @return instance of this class
136         * @throws IOException if establishing the connection fails
137         */
138        public static SocketTransport connect(String host, int port, X509Certificate pinnedcert) throws IOException {
139                try {
140                        // Create a trust manager that does not validate certificate chains
141                        TrustManager[] trustAllCerts = new TrustManager[] {
142                                        new X509TrustManager() {
143                                                @Override
144                                                public X509Certificate[] getAcceptedIssuers() {
145                                                        return new X509Certificate[0];
146                                                }
147                                                @Override
148                                                public void checkServerTrusted(X509Certificate[] arg0, String arg1) throws CertificateException {
149                                                }
150                                                @Override
151                                                public void checkClientTrusted(X509Certificate[] arg0, String arg1) throws CertificateException {
152                                                }
153                                        }
154                        };
155
156                        // Trust managers for SSL context
157                        SSLContext sc = SSLContext.getInstance("TLS");
158
159                        if (pinnedcert != null) {
160                                TrustManagerFactory tmf = TrustManagerFactory.getInstance("X509");
161                                KeyStore ks = null;
162                                ks = KeyStore.getInstance(KeyStore.getDefaultType());
163                                // Generate an empty one
164                                ks.load(null, null);
165                                ks.setCertificateEntry("pinned", pinnedcert);
166                                tmf.init(ks);
167                                sc.init(null, tmf.getTrustManagers(), new java.security.SecureRandom());
168                        } else {
169                                sc.init(null, trustAllCerts, new java.security.SecureRandom());
170                        }
171
172                        // Connect with created parameters
173                        SSLSocketFactory factory = sc.getSocketFactory();
174                        Socket s = factory.createSocket(host, port);
175                        return new SocketTransport(s);
176                } catch (KeyManagementException | KeyStoreException | NoSuchAlgorithmException | CertificateException e) {
177                        throw new IOException("Could not connect", e);
178                }
179        }

181        // Helper class to make a SSL server
182        public static ServerSocket make_server(int port, String pkcs12Path, String pkcs12Pass)
183                        throws IOException, KeyStoreException, NoSuchAlgorithmException, CertificateException, UnrecoverableKeyException, KeyManagementException {
184                return make_server(port, new FileInputStream(pkcs12Path), pkcs12Pass);
185        }
186
187        // Helper class to make a SSL server
188        protected static ServerSocket make_server(int port, InputStream pkcs12stream, String pkcs12Pass)
189                        throws IOException, KeyStoreException, NoSuchAlgorithmException, CertificateException, UnrecoverableKeyException, KeyManagementException {
190                KeyStore ks = KeyStore.getInstance("PKCS12");
191                ks.load(pkcs12stream, pkcs12Pass.toCharArray());
192                KeyManagerFactory kmf = KeyManagerFactory.getInstance(KeyManagerFactory.getDefaultAlgorithm());
193                kmf.init(ks, pkcs12Pass.toCharArray());
194                SSLContext sc = SSLContext.getInstance("TLS");
195                sc.init(kmf.getKeyManagers(), null, null);
196                return sc.getServerSocketFactory().createServerSocket(port);
197        }

200        @Override
201        @SuppressWarnings("unchecked")
202        public synchronized void send(Map<String, Object> msg) throws IOException {
203                JSONObject obj = new JSONObject();
204                obj.putAll(msg);
205                byte[] data = obj.toJSONString().getBytes(Charset.forName("UTF-8"));
206                socket.getOutputStream().write(length.putInt(0, data.length).array());
207                socket.getOutputStream().write(data);
208                logger.debug("> ({}) {}", HexUtils.encodeHexString(length.array()), obj.toJSONString());
209        }
210
211        @Override
212        @SuppressWarnings("unchecked")
213        public synchronized Map<String, Object> recv() throws IOException {
214                if (socket.isClosed()) {
215                        throw new IOException("Connection closed");
216                }
217                length.putInt(0, 0);
218                socket.getInputStream().read(length.array());
219                int len = length.getInt(0);
220                if (len == 0) {
221                        throw new IOException("Failed to read data (length)");
222                }
223                if (len > 1024) {
224                        throw new IOException("Bad message length > 1024");
225                }
226                // Read data
227                byte[] data = new byte[len];
228
229                socket.getInputStream().read(data);
230                JSONObject obj = (JSONObject) JSONValue.parse(new String(data, "UTF-8"));
231                logger.debug("< ({}) {}",  HexUtils.encodeHexString(length.array()), obj.toJSONString());
232                return obj;
233        }
234
235        @Override
236        public void close() {
237                try {
238                        socket.close();
239                } catch (IOException e) {
240                        logger.trace("Could not close socket", e);
241                }
242        }
