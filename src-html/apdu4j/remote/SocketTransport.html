024import java.io.FileInputStream;
025import java.io.IOException;
026import java.net.ServerSocket;
027import java.net.Socket;
028import java.nio.ByteBuffer;
029import java.nio.ByteOrder;
030import java.nio.charset.Charset;
031import java.security.KeyManagementException;
032import java.security.KeyStore;
033import java.security.KeyStoreException;
034import java.security.NoSuchAlgorithmException;
035import java.security.UnrecoverableKeyException;
036import java.security.cert.CertificateException;
037import java.security.cert.X509Certificate;
038import java.util.Map;
039
040import javax.net.ssl.KeyManagerFactory;
041import javax.net.ssl.SSLContext;
042import javax.net.ssl.SSLSocketFactory;
043import javax.net.ssl.TrustManager;
044import javax.net.ssl.TrustManagerFactory;
045import javax.net.ssl.X509TrustManager;
046import javax.xml.bind.DatatypeConverter;
047
048import org.json.simple.JSONObject;
049import org.json.simple.JSONValue;
050
051/**
052 * Based on the <a href=
053 * "https://developer.chrome.com/extensions/nativeMessaging#native-messaging-host-protocol">
054 * NativeMessaging protocol from Chrome</a> (but uses big endian uint32
055 * as is common in network protocols). Messages bigger than 1024 bytes are
056 * rejected by this implementation and the underlying socket is closed.
057 *
058 * @author Martin Paljak
059 */
060public class SocketTransport implements JSONMessagePipe, AutoCloseable {
061        private boolean verbose = false;
062        private final Socket socket;
063        private final ByteBuffer length = ByteBuffer.allocate(Integer.SIZE / Byte.SIZE);
064
065        public SocketTransport(Socket s) {
066                socket = s;
067                length.order(ByteOrder.BIG_ENDIAN);
068        }
069
070        /**
071         * Setting this to true makes the class log to System.out all JSON messages
072         *
073         * @param yes true if the class should be verbose
074         */
075        public void beVerbose(boolean yes) {
076                verbose = yes;
077        }
078
079        /**
080         * Connects to the mentioned host and port with SSL without checking certificate chain.
081         *
082         * @param host host to connect to
083         * @param port port to use
084         * @return instance of this class
085         * @throws IOException if establishing the connection fails
086         */
087        public static SocketTransport connect_insecure(String host, int port) throws IOException {
088                return connect(host, port, null);
089        }
090
091        /**
092         * Connects to the mentioned host and port with SSL and checks the certificate against the pinned version.
093         *
094         * @param host host to connect to
095         * @param port port to use
096         * @param pinnedcert expected certificate of the server
097         * @return instance of this class
098         * @throws IOException if establishing the connection fails
099         */
100        public static SocketTransport connect(String host, int port, X509Certificate pinnedcert) throws IOException {
101                try {
102                        // Create a trust manager that does not validate certificate chains
103                        TrustManager[] trustAllCerts = new TrustManager[] {
104                                        new X509TrustManager() {
105                                                @Override
106                                                public X509Certificate[] getAcceptedIssuers() {
107                                                        return new X509Certificate[0];
108                                                }
109                                                @Override
110                                                public void checkServerTrusted(X509Certificate[] arg0, String arg1) throws CertificateException {
111                                                }
112                                                @Override
113                                                public void checkClientTrusted(X509Certificate[] arg0, String arg1) throws CertificateException {
114                                                }
115                                        }
116                        };
117
118                        // Trust managers for SSL context
119                        SSLContext sc = SSLContext.getInstance("TLS");
120
121                        if (pinnedcert != null) {
122                                TrustManagerFactory tmf = TrustManagerFactory.getInstance("X509");
123                                KeyStore ks = null;
124                                ks = KeyStore.getInstance(KeyStore.getDefaultType());
125                                // Generate an empty one
126                                ks.load(null, null);
127                                ks.setCertificateEntry("pinned", pinnedcert);
128                                tmf.init(ks);
129                                sc.init(null, tmf.getTrustManagers(), new java.security.SecureRandom());
130                        } else {
131                                sc.init(null, trustAllCerts, new java.security.SecureRandom());
132                        }
133
134                        // Connect with created parameters
135                        SSLSocketFactory factory = sc.getSocketFactory();
136                        Socket s = factory.createSocket(host, port);
137                        return new SocketTransport(s);
138                } catch (KeyManagementException | KeyStoreException | NoSuchAlgorithmException | CertificateException e) {
139                        throw new IOException("Could not connect", e);
140                }
141        }
142
143        // Helper class to make a SSL server
144        public static ServerSocket make_server(int port, String pkcs12Path, String pkcs12Pass)
145                        throws IOException, KeyStoreException, NoSuchAlgorithmException, CertificateException, UnrecoverableKeyException, KeyManagementException {
146                KeyStore ks = KeyStore.getInstance("PKCS12");
147                ks.load(new FileInputStream(pkcs12Path), pkcs12Pass.toCharArray());
148                KeyManagerFactory kmf = KeyManagerFactory.getInstance(KeyManagerFactory.getDefaultAlgorithm());
149                kmf.init(ks, pkcs12Pass.toCharArray());
150                SSLContext sc = SSLContext.getInstance("TLS");
151                sc.init(kmf.getKeyManagers(), null, null);
152                return sc.getServerSocketFactory().createServerSocket(port);
153        }
154
155
156        @Override
157        public synchronized void send(Map<String, Object> msg) throws IOException {
158                JSONObject obj = new JSONObject();
159                obj.putAll(msg);
160                byte[] data = obj.toJSONString().getBytes(Charset.forName("UTF-8"));
161                socket.getOutputStream().write(length.putInt(0, data.length).array());
162                socket.getOutputStream().write(data);
163                if (verbose) {
164                        System.out.println("> (" + DatatypeConverter.printHexBinary(length.array()) + ") " + new String(data, "UTF-8"));
165                }
166        }
167
168        @Override
169        public synchronized Map<String, Object> recv() throws IOException {
170                if (socket.isClosed()) {
171                        throw new IOException("Connection closed");
172                }
173                length.putInt(0, 0);
174                socket.getInputStream().read(length.array());
175                int len = length.getInt(0);
176                if (len == 0) {
177                        throw new IOException("Failed to read data (length)");
178                }
179                if (len > 1024) {
180                        throw new IOException("Bad message length > 1024");
181                }
182                // Read data
183                byte[] data = new byte[len];
184
185                socket.getInputStream().read(data);
186                if (verbose) {
187                        System.out.println("< (" + DatatypeConverter.printHexBinary(length.array()) + ") " + new String(data, "UTF-8"));
188                }
189                JSONObject obj = (JSONObject) JSONValue.parse(new String(data, "UTF-8"));
190                return obj;
191        }
192
193        @Override
194        public void close() {
195                try {
196                        socket.close();
197                } catch (IOException e) {}
198        }
